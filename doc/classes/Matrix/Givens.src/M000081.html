                def Givens.givens(a, b)
                        if b == 0 
                                c = 0; s = 0
                        else
                                if b.abs > a.abs
                                        tau = Float(-a)/b; s = 1/Math.sqrt(1+tau**2); c = s * tau
                                else
                                        tau = Float(-b)/a; c = 1/Math.sqrt(1+tau**2); s = c * tau
                                end
                        end
                        return c, s
                end
