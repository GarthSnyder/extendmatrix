    def LU.factorization(mat)
      u = mat.clone 
                        n = u.column_size
                        i = Matrix.I(n)
                      l = i.clone
      (n-1).times {|k| 
                                mk = gauss(u, k)
                                u = mk * u  # M_{n-1} * ... * M_1 * A = U
                                l += i - mk # L = M_1^{-1} * ... * M_{n-1}^{-1} = I + sum_{k=1}^{n-1} tau * e
                        }
                        return l, u
    end
