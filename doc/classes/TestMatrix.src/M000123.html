        def test_eigenvaluesJacobi
                a = Matrix[[1, 1, 1, 4],
                                                         [1, 1, 0, 5],
                                                         [1, 0, 1, 4],
                                                         [4, 5, 4, 1]]
                eigens = Vector[-0.268280385530705, -5.97550058143538, 1.01373431639199, 9.2300466505741]
                assert_in_delta 0, (eigens - a.eigenvaluesJacobi).norm, 1.0e-10
