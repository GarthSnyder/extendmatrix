        def test_diag
                m1 = Matrix[[1]]
                m2 = Matrix[[2, 0], [0, 3]]
                m3 = Matrix[[4, 0, 0], [0, 5, 0], [0, 0, 6]]
                a1 = Matrix.new(6, 6){|i, j| i == j ? i + 1: 0}
                assert_equal a1, Matrix.diag(m1, m2, m3)
                assert_equal m2, Matrix.diag(m2)
                a2 = Matrix[[2, 0, 0, 0],
                                                         [0, 3, 0, 0],
                                                         [0, 0, 2, 0],
                                                         [0, 0, 0, 3]]
                assert_equal a2, Matrix.diag(m2, m2)
        end
