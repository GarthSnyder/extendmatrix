        def test_set
                m = Matrix[[1, 2, 222], [2, 33, 4]]
                n = Matrix.new(2,3)
                n.set(m)
                assert_equal m, n
                m.wrap = :torus
                n.set(m)
                assert_equal :torus, n.wrap
        end
