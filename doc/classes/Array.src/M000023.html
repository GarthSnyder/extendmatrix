        def Array.map(n, *arrays)
                len = arrays.length
                if n == 0
                        n = arrays[0].length
                        1.upto(arrays.length - 1) do |i|
                                al = arrays[i].length
                                n = al if al < n
                        end
                end
                0.upto(n - 1) do |i|
                        params = []
                        0.upto(len - 1) do |arr|
                                params << arrays[arr][i]
                        end
                        yield(*params)
                end
        end
