        def column=(args)
                m = row_size
                c, v, r = MMatrix.id_vect_range(args, m)
                (m..r.begin - 1).each{|i| self[i, c] = 0}
                [v.size, r.entries.size].min.times{|i| self[i + r.begin, c] = v[i]}
                ((v.size + r.begin)..r.entries.last).each {|i| self[i, c] = 0}
        end
