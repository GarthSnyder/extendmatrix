        def to_s(mode = :pretty, len_col = 3)
                return super if empty?
                if mode == :pretty
                        clen = cols_len
                        to_a.collect {|r| mapcar(r, clen) {|x, l| format("%#{l}s ",x.to_s)} << "\n"}.join("")
                else
                        i = 0; s = ""; cs = column_size
                        each do |e|
                                i = (i + 1) % cs
                                s += format("%#{len_col}s ", e.to_s)
                                s += "\n" if i == 0
                        end
                        s
                end
        end
