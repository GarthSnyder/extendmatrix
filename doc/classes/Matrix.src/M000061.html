        def gram_schmidt
                a = clone
                n = column_size
                m = row_size
                q = Matrix.new(m, n){0}
                r = Matrix.zero(n)
                for k in 0...n
                        r[k,k] = a[0...m, k].norm
                        q[0...m, k] = a[0...m, k] / r[k, k]
                        for j in (k+1)...n
                                r[k, j] = q[0...m, k].t * a[0...m, j]
                                a[0...m, j] -= q[0...m, k] * r[k, j]
                        end
                end
                return q, r
        end
