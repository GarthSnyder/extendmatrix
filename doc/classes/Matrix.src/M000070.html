        def cJacobi(tol = 1.0e-10)
                a = self.clone
                n = row_size
                v = Matrix.I(n)
                eps = tol * a.normF
                while Jacobi.off(a) > eps
                        p, q = Jacobi.max(a)
                        c, s = Jacobi.sym_schur2(a, p, q)
                        #print "\np:#{p} q:#{q} c:#{c} s:#{s}\n"
                        j = Jacobi.J(p, q, c, s, n)
                        a = j.t * a * j
                        v = v * j
                end
                return a, v
        end
