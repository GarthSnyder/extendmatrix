        def [](i, j)
                case i
                when Range 
                        case j
                        when Range
                                Matrix[*i.collect{|l| self.row(l)[j].to_a}]
                        else
                                column(j)[i]        
                        end
                else
                        case j
                        when Range
                                row(i)[j]
                        else
                                ids(i, j)
                        end
                end           
        end
