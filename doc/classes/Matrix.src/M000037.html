        def wrap=(mode = :torus)
                case mode
                when :torus then eval(wraplate("i %= row_size; j %= column_size"))
                when :h_cylinder then eval(wraplate("i %= row_size"))
                when :v_cylinder then eval(wraplate("j %= column_size"))
                when :nil then eval(wraplate)
                end
                @wrap = mode
        end
