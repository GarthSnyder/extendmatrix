                def diag(*args)
                        dsize = 0
                        sizes = args.collect{|e| x = (e.is_a?(Matrix)) ? e.row_size : 1; dsize += x; x}
                        m = Matrix.zero(dsize)
                        count = 0

                        sizes.size.times{|i| 
                                range = count..(count+sizes[i]-1)
                                m[range, range] = args[i]
                                count += sizes[i]
                        }
                        m
                end
