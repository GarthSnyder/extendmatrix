        def initialize(*argv)
                return initialize_old(*argv) if argv[0].is_a?(Symbol)
                n, m, val = argv; val = 0 if not val
                f = (block_given?)? lambda {|i,j| yield(i, j)} : lambda {|i,j| val}
                init_rows((0...n).collect {|i| (0...m).collect {|j| f.call(i,j)}}, true)
        end
