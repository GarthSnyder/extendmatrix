        def house
                s = self[1..length-1]
                sigma = s.inner_product(s)
                v = clone; v[0] = 1
                if sigma == 0
                        beta = 0
                else
                        mu = Math.sqrt(self[0] ** 2 + sigma)
                        if self[0] <= 0
                                v[0] = self[0] - mu
                        else
                                v[0] = - sigma.quo(self[0] + mu)
                        end
                        v2 = v[0] ** 2
                        beta = 2 * v2.quo(sigma + v2)
                        v /= v[0]
                end
                return v, beta
        end
