        def Vector.gram_schmidt(*vectors)
                v = vectors.clone
                for j in 0...v.size
                        for i in 0..j-1
                                v[j] -= v[i] * v[j].inner_product(v[i])
                        end
                        v[j] /= v[j].norm
                end
                v
        end
